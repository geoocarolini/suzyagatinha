public class Suzyagatinha extends Robot
{
    public void run(){
        //Cor do Robo
        setBodyColor(new Color(337, 25, 15));

        while(true){
            //Anda para frente 100 
            ahead(300);
            //vire a arma para direita
            turnGunRight(360);
            //Ande para tras 300
            turnGunRight(360);
        }
    }
    //funcao que identifica o robo e faz algo
    public void onScannerdRobot(ScannedRobotEvent e){
        fire(1);//atirar
    }
   //funcao que detecta levar tiro no seu robo
    public void onHitByBullet(HitByBulletEvent e) {
		back(30);//voltar
	}

    public void onHitWall(HitWallEvent e) {
		back(20);//voltar
	}
}
